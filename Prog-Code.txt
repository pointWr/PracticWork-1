{

	cin >> a;

	cin >> c;
	D = pow(b,2)-(4*a*c);
	if (D>0)
	{
		D = sqrt(D);
		x1 = (-b-D)/(2*a);
		x2 = (-b+D)/(2*a);
		cout << "Два корня " << x1 << " & " << x2 << endl;
	}

	{
		x1 = (-b)/(2 * a);
		cout << "Единый корень " << x1 << " " << endl;
	}
	if (D<0)
		cout << "Корней нет " << endl;

}
